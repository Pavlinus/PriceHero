		$('.header-user-menu .navicon').mouseenter(function(){
			if($('.list-curr-menu .has-been-selected.open').length > 0) {
				$('.list-curr-menu .has-been-selected.open').removeClass('open');
				$('.list-curr-menu ul').hide();
			}
			console.log('mouse');
		});
		$('#list-curr-select').customSelectMenu({
    		menuClass : 'list-curr-menu',
    		openedClass : 'open',
    		selectedClass : 'active',
    		selectionMadeClass : 'has-been-selected'
    	});
    	$(document).on('click', '.list-curr-menu li', function() {
    		var	query  = window.location.search.substring(1);
    		window.location.href = location.href+(!query ? '?' : '&')+'sCurr=' + $(this).attr('data-option-value');
    	});


	});
	</script>

	</body>
</html>
